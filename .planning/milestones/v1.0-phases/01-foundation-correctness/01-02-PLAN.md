   ```rust
   // Parse query parameters into (key, value) tuples
   let mut params: Vec<(&str, &str)> = query
       .split('&')
       .filter(|p| !p.is_empty())
       .filter_map(|p| p.split_once('='))
       .filter(|(k, _)| !TRACKING_PARAMS.contains(k))
       .collect();
   // Sort by key name (stable sort preserves order for same-key params)
   params.sort_by_key(|(k, _)| *k);

   if !params.is_empty() {
       let sorted_query: String = params
           .iter()
           .map(|(k, v)| format!("{}={}", k, v))
           .collect::<Vec<_>>()
           .join("&");
       normalized = format!("{}?{}", base_url, sorted_query);
   }
   ```
   Where `TRACKING_PARAMS` is the existing set of tracking parameters to filter out (e.g., `utm_source`, `utm_medium`, etc.). Keep the existing tracking param filter list.

4. Also handle the edge case where a param has no `=` sign (e.g., bare `?flag`): these should be filtered out by `split_once('=')` returning None, which is the desired behavior.

   - Before returning a cached answer, compute the current summary hash: fetch the article's current summary from the `articles` table, hash it with SHA-256 (reuse `generate_content_hash` or use `sha2` directly).
   - Compare with the stored `summary_hash` in the cache row.
   - If they differ (or stored summary_hash is NULL), treat as cache miss and delete the stale cache entry.

   Concrete implementation pattern:
   ```rust
   // After fetching cached row:
   if let Some(cached) = cached_row {
       // Compute current summary hash
       let current_summary: Option<String> = sqlx::query_scalar(
           "SELECT summary FROM articles WHERE id = ?1"
       ).bind(article_id).fetch_optional(db).await?;

       let current_hash = current_summary
           .as_deref()
           .map(|s| {
               use sha2::{Sha256, Digest};
               let mut hasher = Sha256::new();
               hasher.update(s.as_bytes());
               format!("{:x}", hasher.finalize())
           });

       if cached.summary_hash.as_ref() == current_hash.as_ref() {
           return Ok(Some(cached.answer));
       }
       // Cache invalidated -- delete stale entry
       sqlx::query("DELETE FROM deepdive_cache WHERE article_id = ?1 AND question = ?2")
           .bind(article_id).bind(&question)
           .execute(db).await?;
   }
   ```
