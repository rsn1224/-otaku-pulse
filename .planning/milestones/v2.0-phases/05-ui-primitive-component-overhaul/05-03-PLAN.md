     manga: BookOpen,
     game: Gamepad2,
     news: Newspaper,

     contentType: ContentType;
     className?: string;

   export function CoverArtFallback({ contentType, className }: CoverArtFallbackProps): React.JSX.Element {

       <div
         className={cn(
           'w-full h-full rounded-lg flex items-center justify-center',
           `bg-linear-to-br from-(--accent-${contentType}) to-(--surface-container-high)`,
           className,
         )}
       >
         <Icon size={32} className="text-(--on-surface-variant)" aria-hidden="true" />
       </div>
     );
   }
