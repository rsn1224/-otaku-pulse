Replace the entire `batch_interaction_bonuses` function body (lines 64-138) with a single CTE query:

```rust
async fn batch_interaction_bonuses(db: &SqlitePool) -> Result<HashMap<i64, f64>, AppError> {
    let rows: Vec<(i64, f64)> = sqlx::query_as(
        "WITH
          bookmarks AS (
            SELECT id AS article_id, 3.0 AS bonus FROM articles WHERE is_bookmarked = 1 AND is_duplicate = 0
          ),
          deepdives AS (
            SELECT DISTINCT article_id, 1.0 AS bonus FROM article_interactions WHERE action = 'deepdive'
          ),
          dwell_stats AS (
            SELECT article_id,
                   AVG(dwell_seconds) AS avg_dwell,
                   (SELECT AVG(dwell_seconds) FROM article_interactions WHERE dwell_seconds > 0) AS global_avg
            FROM article_interactions WHERE dwell_seconds > 0 GROUP BY article_id
          ),
          feed_engagement AS (
            SELECT a.id AS article_id,
                   CAST(SUM(CASE WHEN ai.action = 'open' THEN 1 ELSE 0 END) AS REAL)
                   / CASE WHEN COUNT(*) = 0 THEN 1 ELSE COUNT(*) END * 1.5 AS bonus
            FROM articles a
            JOIN article_interactions ai ON ai.article_id = a.id
            WHERE a.is_duplicate = 0
            GROUP BY a.id
          )
        SELECT
          a.id AS article_id,
          COALESCE(b.bonus, 0.0) + COALESCE(d.bonus, 0.0) + COALESCE(fe.bonus, 0.0)
          + CASE WHEN ds.avg_dwell > ds.global_avg
                 THEN MIN((ds.avg_dwell - ds.global_avg) / 30.0 * 0.5, 2.0)
                 ELSE 0.0 END AS total_bonus
        FROM articles a
        LEFT JOIN bookmarks b ON b.article_id = a.id
        LEFT JOIN deepdives d ON d.article_id = a.id
        LEFT JOIN dwell_stats ds ON ds.article_id = a.id
        LEFT JOIN feed_engagement fe ON fe.article_id = a.id
        WHERE a.is_duplicate = 0
        ORDER BY a.published_at DESC LIMIT ?",
    )
    .bind(SCORING_ARTICLE_LIMIT)
    .fetch_all(db)
    .await?;

    Ok(rows.into_iter().collect())
}
```

Preserve the `SCORING_ARTICLE_LIMIT` constant and `DWELL_BONUS_CAP` (the 2.0 cap is embedded in the SQL MIN clause).

With a threshold-based parallel/serial approach:
```rust
const RAYON_THRESHOLD: usize = 50;

if articles.len() >= RAYON_THRESHOLD {
    // CPU-bound normalization — parallelize for large batches
    let normalized: Vec<(Option<String>, Option<String>)> = articles
        .par_iter()
        .map(|a| {
            let url_norm = a.url.as_deref().map(dedup_service::normalize_url);
            let hash = a.content.as_deref().map(dedup_service::generate_content_hash);
            (url_norm, hash)
        })
        .collect();
    for (article, (url_norm, hash)) in articles.iter_mut().zip(normalized) {
        article.url_normalized = url_norm;
        article.content_hash = hash;
    }
} else {
    for article in &mut articles {
        if let Some(url) = &article.url {
            article.url_normalized = Some(dedup_service::normalize_url(url));
        }
        if let Some(content) = &article.content {
            article.content_hash = Some(dedup_service::generate_content_hash(content));
        }
    }
}
```

Note: `par_iter()` is used (immutable borrow) with results collected, then applied back to `articles`. This avoids needing `par_iter_mut()` which would require the articles to not be borrowed elsewhere. The threshold of 50 prevents rayon thread pool overhead for small feeds.
