```rust
#[cfg(test)]
mod tests {
    use super::*;
    use tokio_util::sync::CancellationToken;
    use std::time::Duration;

    /// Simulates the core pattern of collect_loop: tokio::select! with CancellationToken
    #[tokio::test]
    async fn test_cancellation_token_exits_loop_immediately() {
        let token = CancellationToken::new();
        let token_clone = token.clone();

        let handle = tokio::spawn(async move {
            loop {
                tokio::select! {
                    _ = token_clone.cancelled() => {
                        break;
                    }
                    _ = tokio::time::sleep(Duration::from_secs(3600)) => {
                        // Would sleep for 1 hour — but cancel should preempt
                    }
                }
            }
        });

        // Cancel immediately
        token.cancel();

        // Loop should exit within 1 second
        let result = tokio::time::timeout(Duration::from_secs(1), handle).await;
        assert!(result.is_ok(), "Loop did not exit within 1 second of cancellation");
        assert!(result.unwrap().is_ok(), "Loop task panicked");
    }

    /// Tests that cancellation during an active sleep exits promptly
    #[tokio::test]
    async fn test_cancellation_during_sleep_exits_promptly() {
        let token = CancellationToken::new();
        let token_clone = token.clone();

        let handle = tokio::spawn(async move {
            tokio::select! {
                _ = token_clone.cancelled() => {
                    return "cancelled";
                }
                _ = tokio::time::sleep(Duration::from_secs(60)) => {
                    return "slept";
                }
            }
        });

        // Wait a moment then cancel
        tokio::time::sleep(Duration::from_millis(50)).await;
        token.cancel();

        let result = tokio::time::timeout(Duration::from_secs(1), handle).await;
        assert!(result.is_ok(), "Task did not exit within 1 second");
        assert_eq!(result.unwrap().unwrap(), "cancelled");
    }

    /// Tests that watch::channel config update breaks the sleep in digest_loop pattern
    #[tokio::test]
    async fn test_config_change_interrupts_sleep() {
        let (config_tx, mut config_rx) = tokio::sync::watch::channel(SchedulerConfig {
            enabled: true,
            interval_minutes: 60,
            digest_hour: 6,
            digest_minute: 0,
        });

        let handle = tokio::spawn(async move {
            tokio::select! {
                result = config_rx.changed() => {
                    return if result.is_ok() { "config_changed" } else { "channel_closed" };
                }
                _ = tokio::time::sleep(Duration::from_secs(3600)) => {
                    return "slept";
                }
            }
        });

        // Send config update
        tokio::time::sleep(Duration::from_millis(50)).await;
        config_tx.send(SchedulerConfig {
            enabled: false,
            interval_minutes: 30,
            digest_hour: 6,
            digest_minute: 0,
        }).unwrap();

        let result = tokio::time::timeout(Duration::from_secs(1), handle).await;
        assert!(result.is_ok(), "Task did not respond to config change within 1 second");
        assert_eq!(result.unwrap().unwrap(), "config_changed");
    }
}
```

    assert!((score - 0.3).abs() < 0.01, "None published_at should return ~0.3 fallback, got {}", score);

#[test]
fn test_calc_base_score_73h_old_article() {
    // 73 hours old — beyond 72h boundary, should get lowest tier score
    let old_date = chrono::Utc::now() - chrono::Duration::hours(73);
    let score = calc_base_score(Some(old_date.to_rfc3339().as_str()));
    // Score for 73h should be lower than 24h score
    let recent_score = calc_base_score(Some(chrono::Utc::now().to_rfc3339().as_str()));
    assert!(score < recent_score, "73h article ({}) should score lower than fresh article ({})", score, recent_score);
}

#[test]
fn test_calc_base_score_unparseable_date() {
    // Invalid date string → should return fallback, not panic
    let score = calc_base_score(Some("not-a-date"));

}
